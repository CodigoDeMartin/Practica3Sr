procedure Mover_3Veces(direcciónAMover){
    /*
    Propósito: Mover el cabezal 3 veces en la dirección $direcciónAMover dada
    Precondición:
     -debe haber almenos 3 celdas en la dirección $direcciónAMover dada
    Paramétros:
     -direcciónAMover es la dirección en la cual el procedimiento se moverá
    */
    repeat (3){
        Mover(direcciónAMover)
    }
}
